------ b. Student - ocena - przedmiot (uczelnia.xml) ------
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE uczelnia [
    <!ELEMENT uczelnia (studenci, przedmioty, oceny)>
    <!ELEMENT studenci (student+)>
    <!ELEMENT student (imie, nazwisko, nr_indeksu)>
    <!ATTLIST student id ID #REQUIRED>
    <!ELEMENT imie (#PCDATA)>
    <!ELEMENT nazwisko (#PCDATA)>
    <!ELEMENT nr_indeksu (#PCDATA)>
    <!ELEMENT przedmioty (przedmiot+)>
    <!ELEMENT przedmiot (nazwa, ects)>
    <!ATTLIST przedmiot id ID #REQUIRED>
    <!ELEMENT nazwa (#PCDATA)>
    <!ELEMENT ects (#PCDATA)>
    <!ELEMENT oceny (ocena+)>
    <!ELEMENT ocena (wartosc, data)>
    <!ATTLIST ocena id ID #REQUIRED>
    <!ATTLIST ocena student_ref IDREF #REQUIRED>
    <!ATTLIST ocena przedmiot_ref IDREF #REQUIRED>
    <!ELEMENT wartosc (#PCDATA)>
    <!ELEMENT data (#PCDATA)>
]>
<uczelnia>
    <studenci>
        <student id="S1"><imie>Adam</imie><nazwisko>Kot</nazwisko><nr_indeksu>1001</nr_indeksu></student>
        <student id="S2"><imie>Ewa</imie><nazwisko>Lis</nazwisko><nr_indeksu>1002</nr_indeksu></student>
    </studenci>
    <przedmioty>
        <przedmiot id="P1"><nazwa>Matematyka</nazwa><ects>6</ects></przedmiot>
        <przedmiot id="P2"><nazwa>Fizyka</nazwa><ects>5</ects></przedmiot>
    </przedmioty>
    <oceny>
        <ocena id="O1" student_ref="S1" przedmiot_ref="P1"><wartosc>4.0</wartosc><data>2025-01-20</data></ocena>
        <ocena id="O2" student_ref="S2" przedmiot_ref="P2"><wartosc>3.5</wartosc><data>2025-01-21</data></ocena>
    </oceny>
</uczelnia>
